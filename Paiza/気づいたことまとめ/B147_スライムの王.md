n, x = map(int, input().split())
a = list(map(int, input().split()))
a.sort()

# 取り込んだ回数
count = 0

# まだ自分より大きいスライムがいる間だけ続ける
while a and x <= a[-1]:
    # x 以下の最大を探す（線形で十分：N<=100）
    j = 0
    while j < len(a) and a[j] <= x:
        j += 1

    if j == 0:
        # 取り込めるスライムが無い → 到達不能
        print(-1)
        break

    # x 以下の最大（a[j-1]）を取り込む
    x += a[j - 1]
    count += 1
    a.pop(j - 1)
else:
    # すでに最大超え、または配列が空になったら回数を出力
    print(count)
